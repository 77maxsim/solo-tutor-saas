2.3 Archiving helpers

In the same data file, add:

export async function archiveStudents(supabase, ids: string[]) {
  if (!ids?.length) return { data: [], error: null };
  return await supabase.from('students')
    .update({ archived_at: new Date().toISOString() })
    .in('id', ids);
}

export async function unarchiveStudents(supabase, ids: string[]) {
  if (!ids?.length) return { data: [], error: null };
  return await supabase.from('students')
    .update({ archived_at: null })
    .in('id', ids);
}

2.4 Safe hard-delete (only if zero sessions) via RPC

Create SQL function (one-shot):

create or replace function public.delete_students_if_no_sessions(p_tutor uuid, p_ids uuid[])
returns table (deleted_id uuid)
language sql
security definer
as $$
  delete from public.students s
  where s.tutor_id = p_tutor
    and s.id = any(p_ids)
    and not exists (select 1 from public.sessions se where se.student_id = s.id)
  returning s.id;
$$;


Client wrapper (e.g., in students.ts):

export async function hardDeleteIfNoSessions(supabase, tutorId: string, ids: string[]) {
  if (!ids?.length) return { data: [], error: null };
  return await supabase.rpc('delete_students_if_no_sessions', {
    p_tutor: tutorId, p_ids: ids
  });
}

3.4 Confirm modals (Archive/Delete) + Undo toast

Add two tiny utilities (can be in this page or a hooks/util file):

import { useToast } from "@/components/ui/use-toast"; // adjust path
const { toast } = useToast();

async function archiveConfirm(ids: string[]) {
  if (!ids.length) return;
  const ok = window.confirm(`Archive ${ids.length} student(s)? They will be hidden from lists but remain in reports.`);
  if (!ok) return;
  const { error } = await archiveStudents(supabase, ids);
  if (error) return toast({ title: "Archive failed", description: error.message, variant: "destructive" });
  toast({
    title: "Archived",
    description: `${ids.length} student(s) archived.`,
    action: {
      label: "Undo",
      onClick: async () => { await unarchiveStudents(supabase, ids); refetch(); }
    } as any
  });
  setSelectedIds([]); refetch();
}

async function deleteConfirm(ids: string[]) {
  if (!ids.length) return;
  const ok = window.confirm(`Delete ${ids.length} student(s)? This cannot be undone. Only students with zero sessions will be deleted.`);
  if (!ok) return;
  const { data, error } = await hardDeleteIfNoSessions(supabase, currentTutorId, ids);
  if (error) return toast({ title: "Delete failed", description: error.message, variant: "destructive" });

  const deleted = data?.length ?? 0;
  const blocked = ids.length - deleted;
  if (deleted) toast({ title: "Deleted", description: `${deleted} student(s) removed.` });
  if (blocked) toast({
    title: "Some not deleted",
    description: `${blocked} had sessions and were kept. Use Archive instead.`,
  });
  setSelectedIds([]); refetch();
}


refetch() refers to your students query reloader (React Query/SWR/custom). Use whatever your page already uses.
