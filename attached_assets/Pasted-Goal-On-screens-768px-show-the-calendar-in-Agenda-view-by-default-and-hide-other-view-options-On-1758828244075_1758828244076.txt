Goal: On screens ≤768px, show the calendar in Agenda view by default and hide other view options. On larger screens, keep existing Week/Month/Agenda.
Scope: Frontend only (React + react-big-calendar). No DB or timezone changes.

Create a tiny hook for responsive checks:

Add src/hooks/useMediaQuery.ts:

import { useEffect, useState } from "react";

export function useMediaQuery(query: string) {
  const [matches, setMatches] = useState(false);
  useEffect(() => {
    const mql = window.matchMedia(query);
    const onChange = (e: MediaQueryListEvent) => setMatches(e.matches);
    // set initial
    setMatches(mql.matches);
    // listen
    if (mql.addEventListener) mql.addEventListener("change", onChange);
    else mql.addListener(onChange);
    return () => {
      if (mql.removeEventListener) mql.removeEventListener("change", onChange);
      else mql.removeListener(onChange);
    };
  }, [query]);
  return matches;
}

import { Calendar, Views } from "react-big-calendar";
import { useMediaQuery } from "@/hooks/useMediaQuery";
// ...existing imports (localizer, events, components, etc.)

export default function CalendarPage() {
  const isMobile = useMediaQuery("(max-width: 768px)");

  // Desktop: keep Week/Month/Agenda. Mobile: Agenda-only.
  const availableViews = isMobile
    ? { agenda: true }
    : { week: true, month: true, agenda: true };

  const defaultView = isMobile ? Views.AGENDA : Views.WEEK;

  return (
    <div className="h-full w-full">
      {/* If you have custom view toggle buttons, hide them on mobile */}
      {!isMobile && (
        <div className="mb-3 flex gap-2">
          {/* existing buttons/tabs for Week / Month / Agenda stay as-is */}
        </div>
      )}

      <Calendar
        // ✅ unchanged timezone/localizer props stay exactly as they are now
        localizer={localizer}
        events={events}
        startAccessor="start"
        endAccessor="end"

        // 🔽 responsive behavior
        views={availableViews}
        defaultView={defaultView}

        // make agenda show a full week list
        length={7}

        // keep your existing props (min/max, step, components, messages, etc.)
        {...otherCalendarProps}
      />
    </div>
  );
}
