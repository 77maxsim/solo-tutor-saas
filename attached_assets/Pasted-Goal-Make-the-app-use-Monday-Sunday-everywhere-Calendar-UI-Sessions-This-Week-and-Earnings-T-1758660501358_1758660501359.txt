Create a single source of truth for week boundaries

Add src/lib/dateRange.ts (or update if it exists):

// src/lib/dateRange.ts
import { startOfWeek, endOfWeek, startOfMonth, endOfMonth } from "date-fns";
import { zonedTimeToUtc, utcToZonedTime } from "date-fns-tz";

// The app-wide setting for week start
export const WEEK_STARTS_ON: 0 | 1 = 1; // 1 = Monday, 0 = Sunday

// Helper to get a user's timezone (fallback to app default)
export const APP_TIMEZONE = Intl.DateTimeFormat().resolvedOptions().timeZone || "Europe/Kyiv";

export function weekRange(date = new Date(), tz = APP_TIMEZONE) {
  const local = utcToZonedTime(date, tz);
  const startLocal = startOfWeek(local, { weekStartsOn: WEEK_STARTS_ON });
  const endLocal = endOfWeek(local, { weekStartsOn: WEEK_STARTS_ON });
  // Convert to UTC for querying the DB (assuming DB stores UTC)
  return {
    startUtc: zonedTimeToUtc(startLocal, tz),
    endUtc: zonedTimeToUtc(endLocal, tz),
  };
}

export function monthRange(date = new Date(), tz = APP_TIMEZONE) {
  const local = utcToZonedTime(date, tz);
  const startLocal = startOfMonth(local);
  const endLocal = endOfMonth(local);
  return {
    startUtc: zonedTimeToUtc(startLocal, tz),
    endUtc: zonedTimeToUtc(endLocal, tz),
  };
}
