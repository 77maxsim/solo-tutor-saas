// AddAvailabilityModal.tsx (core parts only)
import React, { useMemo, useState, useCallback } from 'react';
// import your existing helpers/hooks:
import { localToUtc } from '@/utils/time'; // ← keep using exactly as implemented
import { useToast } from '@/hooks/useToast';
import AvailabilityGrid from './AvailabilityGrid';
import { createAvailabilitySlots } from '@/services/availability'; // new bulk function

type LocalRange = { startLocal: Date; endLocal: Date };

export default function AddAvailabilityModal(props: {
  isOpen: boolean;
  onClose: () => void;
  weekStartLocal: Date; // you already pass this
  bookedRangesLocal: Array<LocalRange>;
  pendingRangesLocal: Array<LocalRange>;
  existingAvailabilityLocal: Array<LocalRange>;
}) {
  const { toast } = useToast();
  const [selectedSlots, setSelectedSlots] = useState<LocalRange[]>([]);

  const overlaps = (a: LocalRange, b: LocalRange) =>
    a.startLocal < b.endLocal && b.startLocal < a.endLocal;

  const collidesWithSystemRanges = (r: LocalRange) => {
    const lists = [
      ...props.bookedRangesLocal,
      ...props.pendingRangesLocal,
      ...props.existingAvailabilityLocal,
    ];
    return lists.some((x) => overlaps(r, x));
  };

  const collidesWithSelected = (r: LocalRange) =>
    selectedSlots.some((x) => overlaps(r, x));

  const normalizeToStep = useCallback((r: LocalRange): LocalRange => {
    // keep your current slot granularity (e.g., 30 min). Reuse existing rounding if you have it.
    // If not, implement floor/ceil to nearest 30 min without changing timezone logic.
    const stepMin = 30;
    const floorToStep = (d: Date) => {
      const t = new Date(d);
      t.setSeconds(0, 0);
      const m = t.getMinutes();
      t.setMinutes(m - (m % stepMin));
      return t;
    };
    const ceilToStep = (d: Date) => {
      const t = new Date(d);
      t.setSeconds(0, 0);
      const m = t.getMinutes();
      const next = m % stepMin === 0 ? m : m + (stepMin - (m % stepMin));
      t.setMinutes(next);
      return t;
    };
    return { startLocal: floorToStep(r.startLocal), endLocal: ceilToStep(r.endLocal) };
  }, []);

  const onProposedRange = (r: LocalRange) => {
    const norm = normalizeToStep(r);
    if (norm.endLocal <= norm.startLocal) {
      toast.warning('Please select a valid time range.');
      return;
    }
    if (collidesWithSystemRanges(norm)) {
      toast.warning('That range overlaps an existing or booked slot.');
      return;
    }
    if (collidesWithSelected(norm)) {
      toast.warning('That range overlaps one you already selected.');
      return;
    }
    // de-dup exact match
    if (
      selectedSlots.some(
        (x) =>
          x.startLocal.getTime() === norm.startLocal.getTime() &&
          x.endLocal.getTime() === norm.endLocal.getTime()
      )
    ) {
      toast.info('This range is already in your selection.');
      return;
    }
    setSelectedSlots((prev) => [...prev, norm].sort((a, b) => +a.startLocal - +b.startLocal));
  };

  const removeSelected = (idx: number) =>
    setSelectedSlots((prev) => prev.filter((_, i) => i !== idx));

  const clearSelection = () => setSelectedSlots([]);

  const onSubmit = async () => {
    if (selectedSlots.length === 0) return;

    // IMPORTANT: preserve existing time-zone conversion pipeline
    const payloads = selectedSlots.map((r) => ({
      start_utc: localToUtc(r.startLocal), // EXACTLY as your single-slot code
      end_utc: localToUtc(r.endLocal),
      // include any additional fields you add today for a single slot:
      // tutor_id, created_by, notes, etc.
    }));

    const { error } = await createAvailabilitySlots(payloads);
    if (error) {
      toast.error('Failed to add some slots. Please try again.');
      return;
    }
    toast.success(`Added ${selectedSlots.length} slot${selectedSlots.length > 1 ? 's' : ''}.`);
    clearSelection(); // keep modal open for more selection
  };

  return (
    <Modal isOpen={props.isOpen} onClose={props.onClose} title="Add Availability Slot">
      <AvailabilityGrid
        weekStartLocal={props.weekStartLocal}
        bookedRangesLocal={props.bookedRangesLocal}
        pendingRangesLocal={props.pendingRangesLocal}
        existingAvailabilityLocal={props.existingAvailabilityLocal}
        selectedRangesLocal={selectedSlots} // highlight in grid
        onProposedRange={onProposedRange}
      />

      {/* Selection summary + controls */}
      <div className="mt-3">
        {selectedSlots.length > 0 ? (
          <ul className="text-sm">
            {selectedSlots.map((r, i) => (
              <li key={i} className="flex items-center gap-2">
                <span>
                  {r.startLocal.toLocaleString([], { hour: '2-digit', minute: '2-digit' })} –{' '}
                  {r.endLocal.toLocaleString([], { hour: '2-digit', minute: '2-digit' })}
                </span>
                <button className="text-red-500 hover:underline" onClick={() => removeSelected(i)}>
                  ×
                </button>
              </li>
            ))}
          </ul>
        ) : (
          <p className="text-sm text-muted-foreground">No ranges selected yet.</p>
        )}
      </div>

      <div className="mt-4 flex gap-2 justify-end">
        <Button variant="secondary" onClick={clearSelection}>
          Clear Selection
        </Button>
        <Button onClick={onSubmit} disabled={selectedSlots.length === 0}>
          Add {selectedSlots.length > 0 ? `${selectedSlots.length} Slot${selectedSlots.length > 1 ? 's' : ''}` : 'Slot'}
        </Button>
        <Button variant="outline" onClick={props.onClose}>
          Close
        </Button>
      </div>
    </Modal>
  );
}
